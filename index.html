import { createClient } from "https://esm.sh/@supabase/supabase-js@2.34.0"; const corsHeaders = { 'Access-Control-Allow-Origin': '*', 'Access-Control-Allow-Methods': 'POST, OPTIONS', 'Access-Control-Allow-Headers': 'authorization, x-client-info, apikey, content-type', 'Content-Type': 'application/json' }; Deno.serve(async (req: Request) => { if (req.method === 'OPTIONS') return new Response('ok', { status: 200, headers: corsHeaders }); try { const sb = createClient(Deno.env.get('SUPABASE_URL')!, Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!, { auth: { persistSession: false } }); // --- PASSO 3: TRAVA DE SEGURANÇA (SEQUENTIAL SAFEGUARD) --- const { data: history } = await sb .from('historico_operacoes') .select('resultado') .order('executado_em', { ascending: false }) .limit(2); // Se as últimas duas operações forem LOSS, bloqueia o motor const isLocked = history && history.length === 2 && history.every(h => h.resultado === 'LOSS'); if (isLocked) { return new Response(JSON.stringify({ status: 'aborted', reason: 'TRAVA DE SEGURANÇA ATIVA: 2 Losses consecutivos detectados. Reset manual necessário.' }), { status: 200, headers: corsHeaders }); } // --- EXECUÇÃO NORMAL DA ENGINE --- const body = await req.json().catch(() => ({})); const symbol = (body.symbol || 'BTCUSDT').toUpperCase(); // Predição de 5 Minutos (Binance) const klinesRes = await fetch(`https://api.binance.com/api/v3/klines?symbol=${symbol}&interval=1m&limit=5`); const klines = await klinesRes.json(); const lastClose = parseFloat(klines[klines.length - 1][4]); // Notícias Financeiras (NewsAPI) const newsRes = await fetch(`https://newsapi.org/v2/everything?q=${symbol.replace("USDT", "")}+market&language=en&sortBy=publishedAt&pageSize=3`, { headers: { 'Authorization': `Bearer ${Deno.env.get('NEWS_API_KEY')}` } }); const newsData = await newsRes.json(); const context = newsData.articles?.map((a: any) => a.title).join(" | ") || "Contexto técnico."; // IA: Geração de Sinal const aiResp = await fetch('https://api.openai.com/v1/chat/completions', { method: 'POST', headers: { 'Authorization': `Bearer ${Deno.env.get('OPENAI_API_KEY')}`, 'Content-Type': 'application/json' }, body: JSON.stringify({ model: 'gpt-4o-mini', messages: [{ role: 'system', content: "Expert Trader. Retorne JSON se houver confluência." }, { role: 'user', content: `Analise ${symbol}. News: ${context}. Preço: ${lastClose}.` }] }) }); const aiJson = await aiResp.json(); const verdict = JSON.parse(aiJson.choices[0].message.content.replace(/```json|```/g, "").trim()); // Persistência na Base v4 await sb.from('log_oportunidades').insert([{ ativo: verdict.ativo, score_confianca: verdict.score, motivo_fato_relevante: verdict.analise, preco_entrada: lastClose.toFixed(2), take_profit: Number(verdict.alvo).toFixed(2), stop_loss: Number(verdict.stop).toFixed(2), fonte: "ENGINE" }]); return new Response(JSON.stringify({ status: 'success' }), { status: 200, headers: corsHeaders }); } catch (err) { return new Response(JSON.stringify({ status: 'aborted', error: err.message }), { status: 200, headers: corsHeaders }); } });
